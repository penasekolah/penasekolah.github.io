[tex] = \frac{362880}{2} [/tex]

i = 1 huruf

g = 1 huruf

t = 1 huruf

[tex]p = \frac{total \: huruf}{huruf \: ganda} [/tex]

p = 9!/2!

p = (9×8×7×6×5×4×3×2×1)/(2×1)

p = 362.880/2
