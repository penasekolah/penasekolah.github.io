[tex] {5}^{3} \times {5}^{4} \div {5}^{12} = {5}^{ - n} [/tex]

[tex] {5}^{3 + 4 - 12} = {5}^{ - n} [/tex]

[tex] - 5 = - n[/tex]

[tex]n = 5[/tex]
