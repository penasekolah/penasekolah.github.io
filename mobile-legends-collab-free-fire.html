[tex]616 = \frac{22}{7} \times {r}^{2} \\ 22 {r}^{2} = 616 \times 7 \\ 22 {r}^{2} = 4.312 \\ {r}^{2} = \frac{4.312}{22 } \\ {r}^{2} = 196[/tex]
