• 3² = 3.3.

• 3³ = 3.3.3

• 3⁴ = 3.3.3.3

[tex] {a}^{n} = a \times a \times a... \times a [/tex]

[tex] {a}^{b} \times {a}^{c} = {a}^{b + c} [/tex]

[tex]( {a}^{b} {)}^{c} = {a}^{bc}[/tex]
