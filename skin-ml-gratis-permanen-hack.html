Un = a + (n - 1)b

U7 = a + (7 - 1)b

15 = a + (6)b

15 = a + 6b

a + 6b = 15 .............. (2)

Eliminasi

a + 2b = 7

a + 6b = 15

___________--

-4b = -8

b = -8/-4

b = 2

Subsitusi

a + 2b = 7

a + 2(2) = 7

a + 4 = 7

a = 7 - 4

a = 3
